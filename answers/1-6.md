# CompareNat1

// CompareNat1である
// 例えば1<4を2通りの方法で導出することを考える
// 最後に1<2, 2<4にL-Transを適用する方法
S(Z) is less than S(S(S(S(Z)))) by L-Trans {
  S(Z) is less than S(S(Z)) by L-Succ {};
  S(S(Z)) is less than S(S(S(S(Z)))) by L-Trans {
    S(S(Z)) is less than S(S(S(Z))) by L-Succ {};
    S(S(S(Z))) is less than S(S(S(S(Z)))) by L-Succ {}
  }
}

// 最後に1<3, 3<4にL-Transを適用する方法
S(Z) is less than S(S(S(S(Z)))) by L-Trans {
  S(Z) is less than S(S(S(Z))) by L-Trans {
    S(Z) is less than S(S(Z)) by L-Succ {};
    S(S(Z)) is less than S(S(S(Z))) by L-Succ {}
  };
  S(S(S(Z))) is less than S(S(S(S(Z)))) by L-Succ {}
}
